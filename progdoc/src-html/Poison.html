002/**
003 * Poison Class - A special Item, when eated it limit the number of remaining moves
004 *
005 * @author Antoine MARMOL
006 * @version 11 / 21/05/2024
007 */
008public class Poison extends ConsumableItem
009{
010    private int aEffect;
011    
012    /**
013     * Constructs a Poison item
014     * 
015     * @param pItemName Name of the Poison item
016     * @param pEffect Effect of the Poison item (nb of remaining moves when consume)
017     */
018    public Poison(final String pItemName,final int pEffect)
019    {
020        super(pItemName, "I shouldn't drink that!", 100, 5.0);
021        this.aEffect = pEffect;
022    }
023    
024    /**
025     * Apply the effect of the player
026     * 
027     * @param pPlayer The player you're applying the effect
028     */
029    public void applyEffect(final Player pPlayer)
030    {
031        pPlayer.setTimeLimiter(true);
032        pPlayer.getTimeLimit().setRemainingMoves(this.aEffect);
033    }
034    
035    /**
036     * Gets the effect of this object
037     * 
038     * @return the effect of this object
039     */
040    public int getEffect()
041    {
042        return this.aEffect;
043    }
044}
