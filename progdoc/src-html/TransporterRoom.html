001import java.util.ArrayList;
002
003/**
004 * TransporterRoom Class - a room that transports the player into a random room of the map
005 *
006 * @author Antoine MARMOL
007 * @version 11 / 21/05/2024
008 */
009public class TransporterRoom extends Room
010{
011    private RoomRandomizer aRoomRandomizer;
012    private Room aRoomAlea;
013    
014    /**
015     * Constructs a TransporterRoom object
016     * 
017     * @param pName name of the room
018     * @param pDes  description of the room
019     * @param pImageName name of the image / path
020     */
021    public TransporterRoom(final String pName, final String pDes, final String pImageName)
022    {
023        super(pName,pDes,pImageName);
024        this.aRoomRandomizer = new RoomRandomizer();
025    }    
026    
027    /**
028     * Puts the exit of the transporter room to a random room
029     * 
030     * @param pRooms the list of rooms created
031     * @return the room which is exit of the transporter room
032     */
033    public Room getExit(final ArrayList<Room> pRooms)
034    {
035        this.aRoomRandomizer.setRandomRoom(pRooms);
036        return this.aRoomRandomizer.getRandomRoom();
037    }
038    
039    /**
040     * Sets the alea room for the alea command
041     * 
042     * @param pRoom the alea room
043     */
044    public void setAleaRoom(final Room pRoom)
045    {
046        this.aRoomAlea = pRoom;
047    }
048    
049    /**
050     * Gets the alea room for the alea command
051     * 
052     * @return pRoom the alea room
053     */
054    public Room getAleaRoom()
055    {
056        return this.aRoomAlea;
057    }
058}
