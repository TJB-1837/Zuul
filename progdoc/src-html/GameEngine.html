319    /**
320     * Given a command, process (that is: execute) the command.
321     * If this command ends the game, true is returned, otherwise false is
322     * returned.
323     * 
324     * @param pCommandLine The command entered by the user.
325     */
326    public void interpretCommand( final String pCommandLine ) 
327    {
328        this.aGui.println( "> " + pCommandLine );
329        Command vCommand = this.aParser.getCommand( pCommandLine );
330
331        if ( vCommand.isUnknown() ) {
332            this.aGui.println( "I don't know what you mean..." );
333            return;
334        }
335        
336        CommandWord vCommandWord = vCommand.getCommandWord();
337        switch(vCommandWord){
338            case GO: this.goRoom(vCommand);
339                break;
340            case HELP: this.printHelp();
341                break;
342            case LOOK: this.look(vCommand);
343                break;
344            case EAT:  this.eat(vCommand);
345                break;
346            case BACK: this.back();
347                break;
348            case INVENTORY: this.inventory();
349                break;
350            case TEST: this.test(vCommand);
351                break;
352            case TAKE: this.take(vCommand);
353                break;
354            case DROP: this.drop(vCommand);
355                break;
356            case FIRE: this.fire(vCommand);
357                break;
358            case CHARGE: this.charge(vCommand);
359                break;
360            case ALEA: this.alea(vCommand);
361                break;
362            case USE:  this.use(vCommand);
363                break;
364            case QUIT:
365                if ( vCommand.hasSecondWord() )this.aGui.println( "Quit what?" );
366                else this.endGame();
367                break;
368            default:
369                this.aGui.println("Programmer error: command not recognised!");
370                break;
371        }
372    }

375    /**
376     * Tests a bunch of commands
377     * 
378     * @param pCommand the document where the commands are
379     */
380    private void test(final Command pCommand)
381    {
382        this.aTestMode = true;
383        if ( !pCommand.hasSecondWord() ) {
384            // if there is no second word, we don't know where to go...
385            this.aGui.println( "Test what? ( indicate the document's name )" );
386            return;
387        }
388        Scanner vDoc; 
389        try{
390            vDoc = new Scanner( new BufferedReader( new FileReader( "Test\\"+ pCommand.getSecondWord() +".txt" ) ) );
391            while(vDoc.hasNextLine()){
392                String vLigne = vDoc.nextLine();
393                interpretCommand(vLigne);
394            }
395            vDoc.close();
396        } 
397        catch( FileNotFoundException Error ){
398            this.aGui.println("This document is not found");
399        }
400    }

402    /**
403     * Saves a room for testing without being worried by the Transporter Room
404     * 
405     * @param pCommand The command entered by the user.
406     */
407    private void alea(final Command pCommand)
408    {
409        if(this.aTestMode == false) return;
410        if(!pCommand.hasSecondWord()){
411            this.aGui.println( "Error: please indicate an index of room." );
412            return;
413        }else{
414            String vIndexS = pCommand.getSecondWord();
415            int vIndex = Integer.parseInt(vIndexS);
416            this.aGui.println("s:"+vIndexS+" i:"+vIndex);
417            for(Room vRoom : this.aRooms){
418                if(vRoom instanceof TransporterRoom){
419                    TransporterRoom vTR = (TransporterRoom)vRoom;
420                    vTR.setAleaRoom( this.aRooms.get(vIndex) );
421                }
422            }
423        }
424    }

534    /** 
535     * Try to go to one direction. If there is an exit, enter the new
536     * room, otherwise print an error message.
537     * 
538     * @param pCommand The command word representing the direction the player wants to go
539     */
540    private void goRoom( final Command pCommand ) 
541    {
542        if ( ! pCommand.hasSecondWord() ) {
543            // if there is no second word, we don't know where to go...
544            this.aGui.println( "Go where?" );
545            return;
546        }
547
548        String vDirection = pCommand.getSecondWord();
549
550        // Try to leave current room.
551        Room vNextRoom;
552        if(this.aPlayer.getCurrentRoom() instanceof TransporterRoom){
553            TransporterRoom vCurrentRoom = (TransporterRoom)this.aPlayer.getCurrentRoom();
554            if(this.aTestMode == true){
555                vNextRoom = vCurrentRoom.getAleaRoom();
556            }else{
557                vNextRoom = vCurrentRoom.getExit( this.aRooms );
558            }
559        }else{
560            vNextRoom = this.aPlayer.getCurrentRoom().getExit( vDirection );
561            if(this.aPlayer.getCurrentRoom().getDoors().containsValue( this.aPlayer.getCurrentRoom().getDoor( vNextRoom ))){
562                String vKeyName = this.aPlayer.getCurrentRoom().getDoor(vNextRoom).getKey().getItemName();
563                if( this.aPlayer.getInventory().containsItem( vKeyName ) ){
564                    this.aPlayer.getCurrentRoom().getDoor(vNextRoom).openDoor( vKeyName );
565                    this.aGui.println( "This door is unlocked." );
566                } else {
567                    this.aPlayer.getCurrentRoom().getDoor(vNextRoom).openDoor( "a" );
568                }
569                if(this.aPlayer.getCurrentRoom().getDoor(vNextRoom).isLocked()){
570                    this.aGui.println( "This door is locked. Find the "+this.aPlayer.getCurrentRoom().getDoor(vNextRoom).getKey().getItemName() );
571                    return; 
572                }
573            }
574        }
575        if ( vNextRoom == null )
576            this.aGui.println( "There is no door!" );
577        else {
578            this.aPlayer.changeRoom(vNextRoom);
579            this.printLocaInfo();
580            if( this.aPlayer.getTimeLimit().getGameOver() ){
581                this.aGui.println("GAME OVER! You lost!");
582                this.endGame();
583            }
584        }
585    }

606     * Returns to the previous room.
607     */
608    private void back()
609    {
610        if(this.aPlayer.isThereAPreviousRoom()){
611            this.aGui.println("There is no previous room.");              
612        } else {
613            Room vNextRoom;
614            if(this.aPlayer.getCurrentRoom() instanceof TransporterRoom){
615                TransporterRoom vCurrentRoom = (TransporterRoom)this.aPlayer.getCurrentRoom();
616                if(this.aTestMode == true){
617                    vNextRoom = vCurrentRoom.getAleaRoom();
618                }else{
619                    vNextRoom = vCurrentRoom.getExit( this.aRooms );
620                }
621                this.aPlayer.changeRoom(vNextRoom);
622                this.printLocaInfo();  
623                return;
624            }
625            this.aPlayer.backToThePreviousRoom();
626            this.printLocaInfo();  
627        }
628    } // back()

631     * Prints out the description of the room and it's image.
632     */
633    private void printLocaInfo()
634    {
635        this.aGui.println( this.aPlayer.getCurrentRoom().getLongDescription() );
636        if ( this.aPlayer.getCurrentRoom().getImageName() != null )
637            this.aGui.showImage( this.aPlayer.getCurrentRoom().getImageName() );
638    }
