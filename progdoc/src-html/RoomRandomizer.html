001import java.util.Random;
002import java.util.ArrayList;
003
004/**
005 * RoomRandomizer Class - finds a random room
006 *
007 * @author Antoine MARMOL
008 * @version 11 / 21/05/2024
009 */
010public class RoomRandomizer
011{
012    private Room aRandomRoom;
013    private Random aRandom;
014    
015    /**
016     * Constructs a RoomRandomizer object
017     */
018    public RoomRandomizer()
019    {
020        this.aRandom = new Random();
021    }
022    
023    /**
024     * creates a random index 
025     * 
026     * @param pNbRooms the number of rooms
027     * @return a random index
028     */
029    public int generateRandomRoomIndex(final int pNbRooms)
030    {
031        return this.aRandom.nextInt(pNbRooms);
032    }
033    
034    /**
035     * Saves a random room base on the size of the list in parameters
036     * 
037     * @param pRooms the ArrayList of created rooms
038     */
039    public void setRandomRoom(final ArrayList<Room> pRooms)
040    {
041        this.aRandomRoom = pRooms.get( this.generateRandomRoomIndex( pRooms.size() ) );
042    }
043    
044    /**
045     * gets the random room stored in this Transporter room
046     * 
047     * @return the saved room
048     */
049    public Room getRandomRoom()
050    {
051        return this.aRandomRoom;
052    }
053}
