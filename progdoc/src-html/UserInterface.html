072    /**
073     * Print out some text into the text area.
074     * 
075     * @param pText The text to be appended to the log.
076     */
077    public void print( final String pText )
078    {
079        this.aLog.append( pText );
080        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
081    } // print(.)
082
083    /**
084     * Print out some text into the text area, followed by a line break.
085     * 
086     * @param pText The text to be appended to the log.
087     */
088    public void println( final String pText )
089    {
090        this.print( pText + "\n" );
091    } // println(.)

093    /**
094     * Show an image file in the interface.
095     * 
096     * @param pImageName The name of the image file to be displayed.
097     */
098    public void showImage( final String pImageName )
099    {
100        String vImagePath = "Images\\"+ pImageName; // to change the directory
101        URL vImageURL = this.getClass().getClassLoader().getResource( vImagePath );
102        if ( vImageURL == null )
103            System.out.println( "Image not found : " + vImagePath );
104        else {
105            ImageIcon vIcon = new ImageIcon( vImageURL );
106            this.aImage.setIcon( new ImageIcon( vIcon.getImage().getScaledInstance(800,800,java.awt.Image.SCALE_SMOOTH) ) );
107        }
108    } // showImage(.)
109
110    /**
111     * Enable or disable input in the entry field and in the buttons.
112     * 
113     * @param pOnOff A boolean value indicating whether to enable (true) or disable (false) user interaction.
114     */
115    public void enable( final boolean pOnOff )
116    {
117        this.aEntryField.setEditable( pOnOff ); // enable/disable
118        this.aEntryField.addActionListener( this );
119        this.aButtonN.setEnabled( pOnOff );
120        this.aButtonS.setEnabled( pOnOff );
121        this.aButtonE.setEnabled( pOnOff );
122        this.aButtonW.setEnabled( pOnOff );
123        this.aButtonBack.setEnabled( pOnOff );
124        this.aButtonInv.setEnabled( pOnOff );
125        this.aButtonAct.setEnabled( pOnOff );
126        this.aButtonUp.setEnabled( pOnOff );
127        this.aButtonDown.setEnabled( pOnOff );
128        if ( pOnOff ) { // enable
129            this.aEntryField.getCaret().setBlinkRate( 500 ); // cursor blink
130            this.aEntryField.addActionListener( this ); // reacts to entry
131        }
132        else { // disable
133            this.aEntryField.getCaret().setBlinkRate( 0 ); // cursor won't blink
134            this.aEntryField.removeActionListener( this ); // won't react to entry
135        }
136    } // enable(.)

268    /**
269     * Actionlistener interface for entry textfield.
270     * 
271     * @param pE The ActionEvent object representing the action event.
272     */
273    @Override public void actionPerformed( final ActionEvent pE ) 
274    {
275        if(this.aButtonN.equals(pE.getSource())){
276            this.aEngine.interpretCommand( "go north" );
277        } else if(this.aButtonS.equals(pE.getSource())){
278            this.aEngine.interpretCommand( "go south" );
279        } else if(this.aButtonE.equals(pE.getSource())){
280            this.aEngine.interpretCommand( "go east" );
281        } else if(this.aButtonW.equals(pE.getSource())){
282            this.aEngine.interpretCommand( "go west" );
283        } else if(this.aButtonBack.equals(pE.getSource())){
284            this.aEngine.interpretCommand( "back" );
285        } else if(this.aButtonInv.equals(pE.getSource())){
286            this.aEngine.interpretCommand( "inventory" );
287        } else if(this.aButtonUp.equals(pE.getSource())){
288            this.aEngine.interpretCommand( "go up" );
289        } else if(this.aButtonDown.equals(pE.getSource())){
290            this.aEngine.interpretCommand( "go down" );
291        } else if(this.aButtonAct.equals(pE.getSource())){
292            this.aEngine.interpretCommand( "quit" );
293        } else {
294            // no need to check the type of action at the moment
295            // because there is only one possible action (text input) :
296            this.processCommand(); // never suppress this line
297        }
298        
299        // no need to check the type of action at the moment
300        // because there is only one possible action (text input) :
301        // this.processCommand(); // never suppress this line
302    } // actionPerformed(.)

304    /**
305     * A command has been entered in the entry field.  
306     * Read the command and do whatever is necessary to process it.
307     */
308    private void processCommand()
309    {
310        String vInput = this.aEntryField.getText();
311        this.aEntryField.setText( "" );
312
313        this.aEngine.interpretCommand( vInput );
314    } // processCommand()
