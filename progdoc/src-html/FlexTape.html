008public class FlexTape extends Item
009{   
010    private int aRepairPercent;
011    
012    /**
013     * Constructs a FlexTape Item
014     * 
015     * @param pName the name of the item
016     * @param pRepairPercent the percentage of reparation the item is able to provide
017     */
018    public FlexTape(final String pName, final int pRepairPercent)
019    {
020        super(pName,"Oh that's a lot of DAMAGE!",5000,40.0);
021        this.aRepairPercent = pRepairPercent;
022    }
023    
024    /**
025     * Gets the percentage of reparability of the item
026     * 
027     * @return the percentage of reparability
028     */
029    public int getRepairPercent()
030    {
031        return this.aRepairPercent;
032    }
033}
