008public class TimeLimit
009{
010    private int aNumberOfMoves;
011    private int aNumberOfRemainingMoves;
012    private boolean aGameOver;
013    private boolean aTimeLimited;
014
015    /**
016     * Constructs a TimeLimit object
017     */
018    public TimeLimit()
019    {
020        this.aNumberOfMoves = 0;
021        this.aGameOver = false;
022        this.aTimeLimited = false;
023    }
024    
025    /**
026     * Checks if the timelimited boolean is true
027     * 
028     * @return the value of time limited status
029     */
030    public boolean isTimeLimited()
031    {
032        return this.aTimeLimited;
033    }
034    
035    /**
036     * Sets if we want to limit in moves or not something
037     * 
038     * @param pTimeLimited The boolean that define the limited status
039     */
040    public void setTimeLimited(final boolean pTimeLimited)
041    {
042        this.aTimeLimited = pTimeLimited;
043    }
044    
045    /**
046     * Gets the number of remaining moves
047     * 
048     * @return the number of remaining moves
049     */
050    public int getRemainingMoves() 
051    {
052        return this.aNumberOfRemainingMoves;
053    }
054    
055    /**
056     * See if the game need to stop because of a time limit exceeding
057     * 
058     * @return if the game is over
059     */
060    public boolean getGameOver() 
061    {
062        return this.aGameOver;
063    }
064    
065    /**
066     * Sets the number of remaining moves
067     * 
068     * @param pRemainingMoves The number of remaining moves
069     */
070    public void setRemainingMoves(final int pRemainingMoves) 
071    {
072        this.aNumberOfRemainingMoves = pRemainingMoves;
073    }
074    
075    /**
076     * Sets the remaining moves for the player and checks if the game should end.
077     * If the remaining moves are zero, the game is set to over.
078     * 
079     * @param pRemainingMoves The number of moves remaining for the player.
080     */
081    public void timeLimit(final int pRemainingMoves)
082    {
083        if(pRemainingMoves == 0) {
084            this.aGameOver=true;
085        }
086        this.setRemainingMoves(pRemainingMoves-1);
087    }
088}
