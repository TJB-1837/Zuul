002/**
003 * RepairableRoom Class - a room that has 2 stages (broken / repaired)
004 *
005 * @author Antoine MARMOL
006 * @version 11 / 21/05/2024
007 */
008public class RepairableRoom extends Room
009{
010    private String aRepairedRoomImagePath;
011    private boolean aRepaired;
012    private int aRepairPercent;
013    
014    /**
015     * Constructs a RepairableRoom object
016     * 
017     * @param pName name of the room
018     * @param pDes description of the room
019     * @param pImageName path of the first image
020     * @param pRepairedRoomImagePath path of the second image
021     */
022    public RepairableRoom(final String pName, final String pDes, final String pImageName,final String pRepairedRoomImagePath)
023    {
024        super(pName,pDes,pImageName);
025        this.aRepairedRoomImagePath = pRepairedRoomImagePath;
026        this.aRepaired = false;
027        this.aRepairPercent = 0;
028    }
029    
030    /**
031     * Sets the repaired status (true or false)
032     * 
033     * @param pRepaired if the room is repaired or not
034     */
035    public void setRepaired(final boolean pRepaired)
036    {
037        this.aRepaired = pRepaired;
038    }
039    
040    /**
041     * Swaps the fist image to the second 
042     */
043    public void changeImageName()
044    {
045        this.setImageName(this.aRepairedRoomImagePath);
046    }
047    
048    /**
049     * Checks if the room is repaired
050     * 
051     * @return If the room is repaired
052     */
053    public boolean isRepaired()
054    {
055        return this.aRepaired;
056    }
057    
058    /**
059     * Gets The percent of repairation of the room
060     * 
061     * @return The percent of repairation of the room
062     */
063    public int getRepairPercent()
064    {
065        return this.aRepairPercent;
066    }
067    
068    /**
069     * Sets the repair percentage of the room
070     * 
071     * @param pRepairPercent the percent of repairation of the room
072     */
073    public void setRepairPercent(final int pRepairPercent)
074    {
075        this.aRepairPercent = pRepairPercent;
076    }
077}
