003/**
004 * Classe Command - une commande du jeu d'aventure Zuul.
005 *
006 * @author Antoine MARMOL
007 * @version 11 / 21/05/2024
008 */
009public class Command
010{
011    private CommandWord aCommandWord;
012    private String aSecondWord = null;
013    
014    // Constructeurs 
015    /**
016     * Constructs a Command object with a specified command word and second word.
017     *
018     * @param pCommandWord The main command word.
019     * @param pSecondWord  The second word associated with the command.
020     */
021    public Command(final CommandWord pCommandWord, final String pSecondWord)
022    {
023        this.aCommandWord = pCommandWord;
024        this.aSecondWord = pSecondWord;
025    }
026    
027    // Accesseurs
028    /**
029     * Gets the command word of the command.
030     *
031     * @return The command word.
032     */
033    public CommandWord getCommandWord()
034    {
035        return this.aCommandWord;
036    }
037    
038    /**
039     * Gets the second word of the command.
040     *
041     * @return The second word, or null if none exists.
042     */
043    public String getSecondWord()
044    {
045        return this.aSecondWord;
046    }
047    
048    // MÃ©thodes 
049    /**
050     * Checks if the command has a second word.
051     *
052     * @return True if a second word is present, false otherwise.
053     */
054    public boolean hasSecondWord()
055    {
056        return this.aSecondWord != null;
057    }
058    
059    /**
060     * Checks if the command is unknown.
061     *
062     * @return True if the command word is null, indicating an unknown command.
063     */
064    public boolean isUnknown()
065    {
066        return this.aCommandWord == CommandWord.UNKNOWN;
067    }
068} // Command
