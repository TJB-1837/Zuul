002/**
003 * Beamer - it's a special item that teleport the player into the room where he charged the beamer
004 *
005 * @author Antoine MARMOL
006 * @version 11 / 21/05/2024
007 */
008public class Beamer extends Item
009{
010    private Room aSavedRoom;
011    
012    /**
013     * Constructs a beamer
014     * 
015     * @param pItemName Name of the Beamer item
016     */
017    public Beamer(final String pItemName)
018    {
019        super(pItemName,"A device that remembers the room you charge it in and teleports you there once you've used it.",50,10.0);
020    }
021    
022    /**
023     * Gets the saved room
024     * 
025     * @return the saved room
026     */
027    public Room getSavedRoom()
028    {
029        return this.aSavedRoom;
030    }
031    
032    /**
033     * Saves the room in the beamer
034     * 
035     * @param pSavedRoom the room you wanna save
036     */
037    public void chargeBeamer(final Room pSavedRoom)
038    {
039        this.aSavedRoom = pSavedRoom;
040    }
041        
042    /**
043     * Verifies if the beamer is charged
044     * 
045     * @return A boolean that return true if there is a saved room
046     */
047    public boolean isCharged()
048    {
049        return this.aSavedRoom != null;
050    }
051    
052    /**
053     * Resets the beamer 
054     */
055    public void resetBeamer()
056    {
057        this.aSavedRoom = null;
058    }
059}
